export const FooterVideo = () => {
    const urlBase = "https://mintlify.s3.us-west-1.amazonaws.com/tonik-a57c11a2/videos/";
    const isLocalHost = window.location.hostname === "localhost" || window.location.hostname === "";
    const url = isLocalHost ? '/videos/extropic-footer' : urlBase + 'extropic-footer';
  
  return useMemo(() => (
    <div 
        id='footer-video'
        key="footer-video"
      className="absolute w-screen pointer-events-none left-1/2 -mt-[100px] sm:ml-[0rem] lg:-ml-[9rem] -translate-x-1/2 z-0 flex items-center justify-center max-h-[720px] overflow-hidden before:bg-black  before:absolute before:inset-0 before:z-1 before:top-0 before:opacity-40"
    >
      <video
        autoPlay
        loop
        muted
        playsInline
        className="w-full h-auto object-cover pointer-events-none"
      >
        <source src={`${url}.mp4`} type="video/mp4" />
        <source src={`${url}.webm`} type="video/webm" />
      </video>
    </div>
  ), []);
};
